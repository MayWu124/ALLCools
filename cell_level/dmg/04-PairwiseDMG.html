import pathlib

Parameters¶

adata_path = '../step_by_step/100kb/adata.with_coords.h5ad'
cluster_col = 'L1'

# change this to the paths to your MCDS files

obs_dim = 'cell'
var_dim = 'gene'

# DMG

                    n_jobs=n_jobs)
